cfa.model <- '
FEE =~ 1*i1 + i2 + i3 + i6
FD =~ 1*i5 + i10 + i11 + i15
FPA =~ NA*i4 + i7 + i9 + i17
FPA ~~ 1*FPA
FEE ~~ FD + FPA
FD ~~ FPA
'

cfa.fit <- sem(cfa.model, sample.cov = cfa.cov, sample.nobs=580)
summary(cfa.fit, fit.measures=T, standardized = T)
